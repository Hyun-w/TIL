# 코드

    #include <string>
    #include <vector>
    #include <algorithm>
    using namespace std;
    int L;
    int R;
    vector<int> solution(int l, int r) {    
        vector<int> answer;
        L=l; R=r;    
        int n=5;
        while(n<=r){
            answer.push_back(n);
            int s=answer.size();
            for(int i=0;i<s-1;i++){
                answer.push_back(n+answer[i]);
            }        
            n*=10;
        }
        
        answer.erase(remove_if(answer.begin(),answer.end(),[](int N){return N<L||N>R;}),answer.end());    
        
        if(answer.empty()) answer.push_back(-1);
        
        return answer;
    }
